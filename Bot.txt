    <script>
        (function() {
            /* ------------------ 1. CONFIG & SETTINGS ------------------ */
            const logoURL = "https://i.ibb.co/fd9wCsn2/icon.jpg"; 
            const backupLogoURL = "https://i.ibb.co/fd9wCsn2/icon.jpg";
            const firebaseURL = "https://talha-trader-admin-panel-lock-default-rtdb.firebaseio.com/users.json";
            const telegramUsername = "@TALHATRADERQUOTEX"; 

            // DOM Elements fetch
            const root = document.getElementById('talha-final-root');
            const qrLine = document.getElementById('qr-scan-line');
            const floatIcon = document.getElementById('floating-bot-icon');

            floatIcon.innerHTML = `<img src="${logoURL}" onerror="this.onerror=null; this.src='${backupLogoURL}';">`;

            // Initial Loading State
            root.innerHTML = `
                <div id="talha-bot">
                    <div class="header-title">TALHA TRADER AUTO BOT</div>
                    <div class="indicator-box">
                        <div class="dot" style="background: #ffcc00;"></div>
                        <span>VERIFYING SYSTEM...</span>
                    </div>
                    <div style="font-size:13px; color:#aaa; margin: 20px 0;">Checking authorization status, please wait...</div>
                </div>
            `;

            /* ------------------ 2. 20-DIGIT ID GENERATION & STORAGE ------------------ */
            let userId = localStorage.getItem('talha_bot_user_id');
            if (!userId) {
                userId = '';
                for (let i = 0; i < 20; i++) {
                    userId += Math.floor(Math.random() * 10).toString();
                }
                localStorage.setItem('talha_bot_user_id', userId);
            }

            /* ------------------ 3. TRADING AUTOMATION LOGIC ------------------ */
            let isProcessing = false; 
            let clickTimer = null;

            function findButtonAndClick(dir) {
                const query = dir === "UP" ? "Up" : "Down";
                // Note: Standalone page par click karne ke liye Quotex platform ka DOM element hona zaroori hai.
                const all = document.querySelectorAll('button, span, div, a');
                for (let el of all) {
                    if (el.innerText.trim() === query && el.offsetWidth > 0) {
                        el.click(); 
                        return true;
                    }
                }
                return false;
            }

            async function startScanning() {
                if (isProcessing) return; 
                isProcessing = true;
                
                qrLine.style.display = "block";
                floatIcon.classList.add('scanning-active');
                const mainTxt = document.getElementById('sig-main');
                const timerTxt = document.getElementById('sig-timer');
                const tf = parseInt(document.getElementById('time-v9').value) || 5;
                const tradeCount = parseInt(document.getElementById('quantity-v9').value) || 1;

                mainTxt.innerText = "SCANNING..."; 
                mainTxt.style.color = "#00ffcc";

                await new Promise(r => setTimeout(r, 4000));
                
                qrLine.style.display = "none"; 
                floatIcon.classList.remove('scanning-active'); 

                const side = Math.random() > 0.5 ? "UP" : "DOWN";
                mainTxt.innerText = side === "UP" ? "UP ↑" : "DOWN ↓";
                mainTxt.style.color = side === "UP" ? "#00ff00" : "#ff3131";

                for (let i = 0; i < tradeCount; i++) {
                    findButtonAndClick(side);
                    if (tradeCount > 1) await new Promise(r => setTimeout(r, 150));
                }

                let remaining = tf;
                let countdown = setInterval(() => {
                    remaining--;
                    timerTxt.innerText = `WAITING: ${remaining}s`;
                    if (remaining <= 0) {
                        clearInterval(countdown);
                        isProcessing = false; 
                        mainTxt.innerText = "IDLE";
                        mainTxt.style.color = "#555";
                        timerTxt.innerText = "READY";
                    }
                }, 1000);
            }

            /* ------------------ 4. INTERFACE RENDERERS ------------------ */
            function renderMainBot() {
                root.innerHTML = `
                    <div id="talha-bot">
                        <div class="header-title">TALHA TRADER AUTO BOT</div>
                        <div class="indicator-box">
                            <div class="dot online"></div>
                            <span>SYSTEM ACTIVE</span>
                        </div>

                        <div id="bot-main">
                            <div class="input-label">TIME</div>
                            <select id="time-v9" class="form-select">
                                <option value="5">5 SECONDS</option>
                                <option value="15">15 SECONDS</option>
                                <option value="30">30 SECONDS</option>
                                <option value="60">1 MINUTE</option>
                                <option value="120">2 MINUTES</option>
                                <option value="300">5 MINUTES</option>
                            </select>

                            <div class="input-label">MARKET</div>
                            <select id="market-v9" class="form-select">
                                <option value="USD-PKR-OTC">USD/PKR (OTC)</option>
                                <option value="USD-MXN-OTC">USD/MXN (OTC)</option>
                                <option value="USD-INR-OTC">USD/INR (OTC)</option>
                                <option value="USD-IDR-OTC">USD/IDR (OTC)</option>
                                <option value="USD-NGN-OTC">USD/NGN (OTC)</option>
                                <option value="USD-PHP-OTC">USD/PHP (OTC)</option>
                                <option value="USD-COP-OTC">USD/COP (OTC)</option>
                                <option value="USD-BRL-OTC">USD/BRL (OTC)</option>
                                <option value="USD-BDT-OTC">USD/BDT (OTC)</option>
                                <option value="USD-ARS-OTC">USD/ARS (OTC)</option>
                            </select>

                            <div class="input-label">NUMBER OF TRADES</div>
                            <input type="number" id="quantity-v9" class="form-input" value="1" min="1" max="20">
                            
                            <div id="sig-display">
                                <div id="sig-main" style="color:#555;">IDLE</div>
                                <div id="sig-timer">READY TO SCAN</div>
                            </div>
                            
                            <button class="action-btn" style="background:#1a1a1a; color:#00ffcc; border:1px solid #333;" id="hide-to-icon">SWITCH TO ICON</button>
                        </div>
                    </div>
                `;

                // Event Handlers
                document.getElementById('hide-to-icon').onclick = () => {
                    root.style.display = "none";
                    floatIcon.style.display = "flex";
                };

                floatIcon.onclick = () => {
                    if (clickTimer == null) {
                        clickTimer = setTimeout(() => {
                            clickTimer = null;
                            startScanning(); 
                        }, 300);
                    } else {
                        clearTimeout(clickTimer);
                        clickTimer = null;
                        floatIcon.style.display = "none";
                        root.style.display = "flex";
                    }
                };
            }

            function renderLockScreen() {
                root.innerHTML = `
                    <div id="talha-bot">
                        <div class="header-title" style="background: linear-gradient(to right, #ff3131, #ff9900); -webkit-background-clip: text; -webkit-text-fill-color: transparent;">DEVICE LOCKED</div>
                        <div class="indicator-box">
                            <div class="dot locked"></div>
                            <span>NOT AUTHORIZED</span>
                        </div>
                        
                        <div style="font-size: 13px; color: #bbb; margin-bottom: 10px; line-height: 1.4;">
                            Your 20-ID copy it and send to the administrator.
                        </div>

                        <div class="input-label" style="color: #ff3131;">YOUR UNIQUE ID:</div>
                        <div class="id-display-box">${userId}</div>

                        <button class="action-btn" style="background: linear-gradient(135deg, #722082, #00ffcc); color: #000; font-weight:900;" id="contact-admin-btn">CONTACT ON TELEGRAM</button>
                    </div>
                `;

                document.getElementById('contact-admin-btn').onclick = () => {
                    window.open(`https://t.me/${telegramUsername.replace('@', '')}`, '_blank');
                };
            }

            /* ------------------ 5. FIREBASE AUTH VERIFICATION ------------------ */
            async function checkAuthorization() {
                try {
                    const response = await fetch(firebaseURL);
                    if (!response.ok) throw new Error("Database error");
                    const data = await response.json();
                    
                    let isAuthorized = false;

                    if (data) {
                        if (typeof data === 'object') {
                            if (data[userId] !== undefined || Object.values(data).includes(userId) || Object.values(data).some(u => u && (u.id === userId || u.userId === userId))) {
                                isAuthorized = true;
                            }
                        }
                    }

                    if (isAuthorized) {
                        renderMainBot();
                    } else {
                        renderLockScreen();
                    }
                } catch (error) {
                    console.error("Auth verification failed: ", error);
                    renderLockScreen();
                }
            }

            // Run verification loop
            checkAuthorization();

        })();
